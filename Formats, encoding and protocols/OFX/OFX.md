Open Financial Exchange

```sgml
<OFX>
<BANKMSGSRSV1>

<!-- Start bank account -->
<STMTTRNRS>
<!-- ... -->
<STMTRS>

<BANKACCTFROM>
<!-- ... -->
</BANKACCTFROM>

<!-- Start list of transactions -->
<BANKTRANLIST>

<!-- Start transaction -->
<STMTTRN>
<!-- ... -->
</STMTTRN>
<!-- End transaction -->

</BANKTRANLIST>
<!-- End list of transactions -->

</STMTRS>
</STMTTRNRS>
<!-- End bank account -->

</BANKMSGSRSV1>
</OFX>
```
